def rotate_list(nums,k):
    if not nums or k==0:
        return nums
    n=len(nums)
    k=k%n
    return nums[-k:] + nums[:-k]

def target_sum(nums, target):
    
    pairs = []
    used = []
    
    for x in nums:
        y = target - x
        
        if x not in used and y in nums:
            
            # handle cases like (5,5)
            if y != x or nums.count(x) > 1:
                
                pairs.append((x, y))
                used.append(x)
                used.append(y)

    return pairs


nums = [2, 4, 3, 5, 7, 8, 9]
target = 10

print(target_sum(nums, target))



def freq(nums):
    counts = {}   # dictionary to store frequencies
    
    for x in nums:
        if x in counts:
            counts[x] += 1    # already seen → increase count
        else:
            counts[x] = 1     # first time → set to 1
            
    return counts


nums = [1, 2, 3, 2, 4, 3]
print(freq(nums))

def remove_ele(nums, value):
    result = []

    for sub in nums:           # each sublist
        new_sub = []           # filtered sublist
        
        for x in sub:          # each element inside sublist

                new_sub.append(x)

    return result

def find_missing(nums):
    n = len(nums)                 # how many numbers are present
    for i in range(1, n + 2):     # expected numbers = 1 to n+1
        if i not in nums:         # if a number is NOT in the list
            return i              # that is the missing number


nums = [1, 2, 4, 5]
print(find_missing(nums))

def second_small(nums):
    smallest=nums[0]
    for x in nums:
        if x<smallest:
            smallest=x
    second = None
    for x in nums:
        if  x != smallest:
              if second is None or x<second:
                     second = x
                
    return second
